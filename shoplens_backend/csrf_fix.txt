
# ---------- CSRF FIX FOR JWT ENDPOINTS ----------
from django.utils.deprecation import MiddlewareMixin

class DisableCSRFForJWT(MiddlewareMixin):
    def process_request(self, request):
        if request.path.startswith('/api/token/') or request.path.startswith('/api/token/refresh/'):
            setattr(request, '_dont_enforce_csrf_checks', True)

MIDDLEWARE.insert(0, 'shoplens_ai.settings.DisableCSRFForJWT')
